def tri_angles(a,b,c):
    tmp = (b**2 + c**2 - a**2) / (2*b*c)
    tmp = np.clip(tmp, -1, 1)
    A = np.arccos(tmp)
    tmp = (a**2 + c**2 - b**2) / (2*a*c)
    tmp = np.clip(tmp, -1, 1)
    B = np.arccos(tmp)
    C = np.pi - (A+B)
    return A,B,C

A,B,C = tri_angles(20,80,80)
print (A,B,C)
